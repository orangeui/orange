# Input

An input component is used to <mark>elicit a response from a user</mark>.

The Input comes in <mark>small (default)</mark> and <mark>large size</mark>, and can hold other components like Buttons when needed.

<hr/>

# Tryout

Tryout is for live example of Input on **Canvas** tab and bottom tab **Knobs**

<Story name='Tryout'>
  {{
    components: { OFormInput, OButton },
    template: `<form class="form" :class="size_class">
      <o-form-input id="form_basic" 
        v-model="form_model" 
        :label="label"
        :description="description"
        :disabled="disabled"
        :read_only="read_only"
        :action_position="action_position">
        <div slot="action">
          <o-button text="copy to clipboard" size="small" />
        </div>
      </o-form-input>
    </form>`,
    props: {
      label: {
        default: text('Label', 'Label')
      },
      description: {
        default: text('Description', 'Description text')
      },
      disabled: {
        default: boolean('Disabled', false)
      },
      read_only: {
        default: boolean('Readonly', false)
      },
      action_position: {
        default: select('Action position', ['hide', 'left', 'right'], 'left')
      },
      form_size: {
        default: select('Size (apply to form class)', ['', 'large'], '')
      },
    },
    data() {
      return {
        form_model: null,
      }
    },
    computed: {
      size_class() {
        return { [`form--${this.form_size}`]: this.form_size }
      }
    }
  }}
</Story>
