# Image

Add styling to images.

<hr/>


# Tryout

Tryout is for live example of Image on **Canvas** tab and bottom tab **Knobs**

<Story name='Tryout'>
  {{
    components: { OImage },
    template: `<div>
      <o-image :path="path" :alt_text="alt_text" :shadow="shadow" :rounded="rounded" :action="action" />
      <a href="#">
        <o-image :path="path2" :alt_text="alt_text" :shadow="shadow" :rounded="rounded" :action="action" />
      </a>
    </div>`,
    props: {
      path: {
        default: text('Image path', 'https://orangecss.com/images/thumbnails/cat.jpg')
      },
      path2: {
        default: text('Image path 2', 'https://orangecss.com/images/thumbnails/square/dog-square.jpg')
      },
      alt_text: {
        default: text('Alt text', 'Katzze')
      },
      shadow: {
        default: boolean('Drop shadow', false)
      },
      rounded: {
        default: select('Rounded', ['-', 'rounded', 'circle'], '-')
      },
      action: {
        default: boolean('Action', false)
      },
    },
  }}
</Story>
