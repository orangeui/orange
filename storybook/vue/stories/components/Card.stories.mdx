# Tryout

Tryout is for live example of Card on **Canvas** tab and bottom tab **Knobs**

<Story name='Tryout'>
  {{
    components: { OCard, OCardHeader, OCardBody, OCardFooter, OButton, OIcon },
    template: `<o-card :full="full">
      <o-card-header>
        <h3>Title</h3>
        <o-button text="Action" v-if="button" />
      </o-card-header>
      <o-card-body>
        Body
      </o-card-body>
      <o-card-footer :action="action" @click="handleClick">
        Footer
        <o-icon symbol="plus" v-if="action"></o-icon>
      </o-card-footer>
    </o-card>`,
    methods: { handleClick: action('click') },
    props: {
      full: {
        default: boolean('Full - no gutter on body', false)
      },
      action: {
        default: boolean('Action on footer with icon', false)
      },
      button: {
        default: boolean('Display button in header', false)
      },
    },
  }}
</Story>
