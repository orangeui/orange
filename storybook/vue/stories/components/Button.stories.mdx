# Button

<hr/>


# Tryout

Tryout is for live example of Button on **Canvas** tab and bottom tab **Knobs**

<Story name='Tryout'>
  {{
    components: { OButton },
    template: `<div :class="buttonExtraClass">
      <o-button @click="handleClick" :type="type" :text="text" :role="role" :icon="icon" :color="color" :size="size" :wide="wide" :inverted="inverted" :is_disabled="is_disabled"></o-button>
    </div>`,
    computed: {
      buttonExtraClass() {
        return {
          [`bg-${this.color}`]: this.inverted,
          [`p-3`]: this.inverted
        }
      }
    },
    methods: { handleClick: action('click') },
    props: {
      type: {
        default: text('Type', 'button')
      },
      text: {
        default: text('Text', 'Hello Storybook')
      },
      role: {
        default: select('Role', 
          { 
            Primary: 'primary',
            Secondary: 'secondary',
            Basic: 'basic',
          }
        )
      },
      icon: {
        default: select('Icon', IconList, IconList[0])
      },
      color: {
        default: select('Color', ColorBase, ColorBase[0])
      },
      size: {
        default: select('Size', 
          { 
            Small: 'small',
            Regular: 'regular',
            Large: 'large',
          }
        )
      },
      wide: {
        default: boolean('Wide', false)
      },
      inverted: {
        default: boolean('Inverted', false)
      },
      is_disabled: {
        default: boolean('Disabled', false)
      }
    },
  }}
</Story>
