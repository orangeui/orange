# Tryout

Tryout is for live example of Textarea on **Canvas** tab and bottom tab **Knobs**

<Story name='Tryout'>
  {{
    components: { OFormTextarea },
    template: `<form class="form" :class="size_class">
      <o-form-textarea id="form_basic" 
        v-model="form_model" 
        :label="label"
        :description="description"
        :disabled="disabled"
        :read_only="read_only"
        :rows="rows"
        :autoexpand="autoexpand">
      </o-form-textarea>
    </form>`,
    props: {
      label: {
        default: text('Label', 'Label')
      },
      description: {
        default: text('Description', 'Description text')
      },
      disabled: {
        default: boolean('Disabled', false)
      },
      read_only: {
        default: boolean('Readonly', false)
      },
      rows: {
        default: number('Rows', 3)
      },
      autoexpand: {
        default: boolean('Auto expand size', false)
      },
      form_size: {
        default: select('Size (apply to form class)', ['', 'large'], '')
      },
    },
    data() {
      return {
        form_model: null,
      }
    },
    computed: {
      size_class() {
        return { [`form--${this.form_size}`]: this.form_size }
      }
    }
  }}
</Story>
