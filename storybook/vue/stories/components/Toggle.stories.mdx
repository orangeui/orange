# Toggle

A Toggle allows users to <mark>decide on a state of an object</mark>.


<hr/>

# Tryout

Tryout is for live example of Toggle on **Canvas** tab and bottom tab **Knobs**

<Story name='Tryout'>
  {{
    components: { OFormToggle },
    template: `<form class="form">
      <o-form-toggle 
        v-model="toggle_model"
        id="toggle_tryout"
        :label="label"
        :description="description"
        :disabled="disabled"
        :size="size"
        :color="color"
        @input="handleClick" />
    </form>`,
    methods: { handleClick: action('change') },
    props: {
      label: {
        default: text('Label', 'Label')
      },
      description: {
        default: text('Description', 'Description text')
      },
      disabled: {
        default: boolean('Disabled', false)
      },
      size: {
        default: select('Size', ['', 'large'], '')
      },
      valueText: {
        default: text('Value text (for size large)', 'Enabled')
      },
      valueTextDisabled: {
        default: text('Value text disabled (for size large)', 'Disabled')
      },
      color: {
        default: select('Color', ColorBase, ColorBase[1])
      }
    },
    data() {
      return {
        toggle_model: false,
      }
    },
  }}
</Story>
