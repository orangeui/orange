# Navigation

A navigation enables users to <mark>move around an app</mark>, visiting areas that are currently not in view.

The component is a custom Vue component written in semantic markup, and comes in <mark>default</mark>, <mark>with Icons</mark>, and <mark>toned in color</mark> styles.

Tryout is for live example of navigation on **Canvas** tab and bottom tab **Knobs**

    template: `<div>

          <o-navigation-item to="/" :icon="navigationItem1Icon" :disabled="disabled" :tag="tag">Navigation</o-navigation-item>
          <o-navigation-item to="/breadcrumbs" :tag="tag">Breadcrumbs</o-navigation-item>
          <o-navigation-item to="/message" :tag="tag">Message</o-navigation-item>
          <o-navigation-item to="/empty-state" :tag="tag">Empty State</o-navigation-item>
        </o-navigation>
        <router-view/>
      </div>`,
    props: {
      color: {
        default: select('Color', ColorBase, null)
      },
      align: {
        default: select('Align', ['left', 'center', 'right'], 'left')
      },
      navigationItem1Icon: {
        default: select('Navigation Item 1 Icon', IconList, 'integrations')
      },
      disabled: {
        default: boolean('Disabled Navigation Item 1', false)
      },
      tag: {
        default: text('Tag (use a or router-link)', 'a')
      }
    }
  }}
</Story>
